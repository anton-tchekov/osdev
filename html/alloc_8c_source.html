104void *memalloc(u32 size)
105{
106 ChunkHeader *cur, *prev;
107 if(size == 0)
108 {
109 return NULL;
110 }
111
112 /* Minimum allocation size */
113 if(size < MIN_ALLOC)
114 {
115 size = MIN_ALLOC;
116 }
117
118 /* Ensure alignment */
119 size = ((size + ALIGN - 1) / ALIGN) * ALIGN;
120
121 _info.Used += size;
122 _info.Total += size + sizeof(ChunkHeader);
123
124 /* Find free chunk */
125 for(prev = &_first, cur = _get_next(prev->Next);
126 cur; prev = cur, cur = _get_next(cur->Next))
127 {
128 if(size + sizeof(ChunkHeader) == cur->Size)
129 {
130 /* --- Perfect fit --- */
131 /* Remove chunk from free list */
132 prev->Next = cur->Next;
133
134 /* Return pointer to data area */
135 return _chunk_data(cur);
136 }
137 else if(size + 2 * sizeof(ChunkHeader) < cur->Size)
138 {
139 /* --- First fit --- */
140 /* Split chunk in two */
141 ChunkHeader *second =
142 (ChunkHeader *)((ptr)cur + sizeof(ChunkHeader) + size);
143
144 second->Size = cur->Size - sizeof(ChunkHeader) - size;
145 second->Next = cur->Next;
146 prev->Next = _get_offset(second);
147
148 /* Reduce the size of first part by the size of the second part */
149 cur->Size = size + sizeof(ChunkHeader);
150
151 /* Return pointer to data area of second part */
152 return _chunk_data(cur);
153 }
154 }
155
156 /* Very unlikely */
157 return NULL;

163
164 header = _chunk_header(p);
165

170 next; prev = next, next = _get_next(next->Next))
171 {
172 if(next > header)
173 {
174 /* Merge chunks if possible */
175 bool mergePrev = ((ptr)prev + prev->Size == (ptr)header);
176 bool mergeNext = ((ptr)header + header->Size == (ptr)next);
177
178 if(mergeNext && mergePrev)
179 {
180 /* Both Prev and Next are free */
181 prev->Next = next->Next;
182 prev->Size += header->Size + next->Size;
183 }
