71bool overlap(Rectangle *c1, Rectangle *c2)
72{
73 /*
74 * The rectangles don't overlap if
75 * one rectangle's minimum in some dimension
76 * is greater than the other's maximum in
77 * that dimension.
78 */
79
80 i32 c1x2 = c1->X + c1->W;
81 i32 c1y2 = c1->Y + c1->H;
82 i32 c2x2 = c2->X + c2->W;
83 i32 c2y2 = c2->Y + c2->H;
84
85 bool noOverlap = c1->X > c2x2 ||
86 c2->X > c1x2 ||
87 c1->Y > c2y2 ||
88 c2->Y > c1y2;
89
90 return !noOverlap;
91}
