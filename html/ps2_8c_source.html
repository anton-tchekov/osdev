172void ps2_init(void)

174 /* Enable external interrupt 0 (PORTB2) */
175 EIMSK |= (1 << PS2_EXT_INTERRUPT);
176
177 /* Trigger interrupt on falling edge */
178 EICRA |= (1 << ISC01);

180 log_boot_P(LOG_INIT, PSTR("PS/2 keyboard initialized"));

407static inline void _ps2_clock_falling_edge(void)
408{
409 if(_bit == 0)
410 {
411 /* Start bit, reset byte */
412 _byte = 0;
413 }
414 else if(_bit >= 1 && _bit <= 8)
415 {
416 /* Data bit */
417 if(_ps2_data_read())
418 {
419 /* Set bit */
420 _byte |= (1 << (_bit - 1));
421 }
422 }
423 else if(_bit == 9)
424 {
425 /* Ignore parity bit */
426 /* TODO: Check parity bit */
427 ps2_event(_byte);
428 }
429 else if(_bit == 10)
430 {
431 /* Stop bit, reset bit counter */
432 _bit = 0;
433
434 /* Return immediately so it stays 0 */
435 return;
436 }
437
438 ++_bit;
439}
