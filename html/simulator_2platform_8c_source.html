129static u32 gfx_color(u8 r, u8 g, u8 b)

137static inline u8 _abgr_r(u32 color) { return (color >> 24) & 0xFF; }

149static u32 _color_merge(u32 color1, u32 color2, u32 ratio)
150{
151 u8 r1, g1, b1, r2, g2, b2;
152
153 r1 = _abgr_r(color1);
154 g1 = _abgr_g(color1);
155 b1 = _abgr_b(color1);
156
157 r2 = _abgr_r(color2);
158 g2 = _abgr_g(color2);
159 b2 = _abgr_b(color2);
160
161 return gfx_color(
162 (r1 * ratio + r2 * (255 - ratio)) / 255,
163 (g1 * ratio + g2 * (255 - ratio)) / 255,
164 (b1 * ratio + b2 * (255 - ratio)) / 255);
165}

167static u32 _rgb565_to_bgra(u16 color)
168{
169 u32 r, g, b;
170 r = (((((u32)color >> 11) & 0x1F) * 527) + 23) >> 6;
171 g = (((((u32)color >> 5) & 0x3F) * 259) + 33) >> 6;
172 b = ((((u32)color & 0x1F) * 527) + 23) >> 6;
173 return gfx_color(r, g, b);
174}

256void env_gfx_image_1bit(
257 i32 x, i32 y, i32 w, i32 h, u32 addr, u32 fg, u32 bg)
258{
259 u8 byte, stride;
260 i32 x0, y0, byte_offset, bit_mask;
261 u8 *image;
262
263 image = (u8 *)(_memory + addr);
264 fg = _abgr_to_argb(fg);
265 bg = _abgr_to_argb(bg);
266 stride = (w + 7) / 8;
267 for(y0 = y; y0 < y + h; ++y0)
268 {
269 bit_mask = 0x80;
270 byte_offset = 0;
271 for(x0 = x; x0 < x + w; ++x0)
272 {
273 if(bit_mask == 0x80)
274 {
275 byte = image[byte_offset++];
276 bit_mask = 1;
277 }
278
279 _pixels[y0 * GFX_WIDTH + x0] = (byte & bit_mask) ? fg : bg;
280 bit_mask <<= 1;
281 }
282
283 image += stride;
284 }

324static Key _convert_key(i32 scancode, i32 mod)
325{
326 Key key = scancode;

358int main(int argc, char **argv)
359{
360 bool running = true;
361 size_t len, offset;
362 FILE *fp;
363
364 if(argc != 2)
365 {
366 fprintf(stderr, "Usage: ./emu rv32i-binary\n");
367 return 1;
368 }
369
370 /* Load binary into memory */
371 if(!(fp = fopen(argv[1], "rb")))
372 {
373 fprintf(stderr, "Failed to open file \"%s\"\n", argv[1]);
374 return 1;
375 }
376
377 offset = 0;
378 do
379 {
380 len = fread(_memory + offset, 1, READ_SIZE, fp);
381 offset += len;
382 }
383 while(len == READ_SIZE);
384
385 timer_init();
386 gfx_init();
387 kernel_init();
388
389 while(running)
390 {
391 SDL_Event e;
392 while(SDL_PollEvent(&e))
393 {
394 switch(e.type)
395 {
396 case SDL_QUIT:
397 running = false;
398 break;
399
400 case SDL_KEYDOWN:
401 {
402 Key key;
403 if(e.key.keysym.sym == SDLK_ESCAPE)
404 {
405 running = false;
406 break;
407 }
408
409 key = _convert_key(e.key.keysym.scancode, e.key.keysym.mod);
410 keyboard_event(key, key_to_codepoint(key),
411 e.key.repeat ? KEYSTATE_REPEAT : KEYSTATE_PRESSED);
412 }
413 break;
414
415 case SDL_KEYUP:
416 {
417 Key key = _convert_key(e.key.keysym.scancode, e.key.keysym.mod);
418 keyboard_event(key, key_to_codepoint(key), KEYSTATE_RELEASED);
419 }
420 break;
421 }
422 }
423
424 os_update();
425
426 SDL_UpdateTexture(_framebuffer, NULL, _pixels, GFX_WIDTH * sizeof(u32));
427 SDL_RenderClear(_renderer);
428 SDL_RenderCopy(_renderer, _framebuffer, NULL, NULL);
429 SDL_RenderPresent(_renderer);
430 }
431
432 gfx_destroy();
433 return 0;
434}
