81static Status _find_entry(
82 const char *name, u32 name_len, u32 block, DirEntry *entry)
83{
84 u8 buf[ATFS_BLOCK_SIZE];
85 u32 i, num_entries, offset;
86
87 RETURN_IF(disk_read(block, buf));
88 num_entries = load_le_32(buf + ATFS_DIR_OFFSET_NUM_ENTRIES);
89 offset = ATFS_DIR_ENTRY_SIZE;
90 for(i = 0; i < num_entries; ++i)
91 {
92 if(offset == ATFS_BLOCK_SIZE)
93 {
94 offset = 0;
95 ++block;
96 RETURN_IF(disk_read(block, buf));
97 }
98
99 dir_entry_get(buf + offset, entry);
100
101 /* Compare name with each directory entry */
102 if(strlen(entry->Name) == name_len &&
103 !strncmp(name, entry->Name, name_len))
104 {
105 return STATUS_OK;
106 }
107
108 offset += ATFS_DIR_ENTRY_SIZE;
109 }
110
111 return STATUS_NO_SUCH_FILE_OR_DIR;
112}

212bool path_valid(const char *path)
213{
214 bool was_sep;
215 char c;
216 const char *s;
217
218 /* Initialize with true because path may not start with separator */
219 was_sep = true;
220 for(s = path; (c = *s); ++s)
221 {
222 if(c == ATFS_DIR_SEPARATOR)
223 {
224 if(was_sep)
225 {
226 /* Two separators in a row are not allowed */
227 return false;
228 }
229
230 was_sep = true;
231 }
232 else if(isdigit(c))
233 {
234 if(was_sep)
235 {
236 /* Path component may not start with a digit */
237 return false;
238 }
239
240 was_sep = false;
241 }
242 else if(c == '_' || islower(c))
243 {
244 was_sep = false;
245 }
246 else
247 {
248 /* Invalid character */
249 return false;
250 }
251 }
252
253 /* Path may not end with separator, but allow empty path */
254 return !was_sep || *path == '\0';
255}

284Status fs_format(void)
285{
286 u32 size;
287 u8 buf[ATFS_BLOCK_SIZE];
288
289 RETURN_IF(disk_size(&size));
290
291 /* Boot Sector */
292 memcpy(buf + ATFS_OFFSET_SIGNATURE, _atfs_signature,
293 sizeof(_atfs_signature));
294
295 write_le_32(buf + ATFS_OFFSET_REVISION, ATFS_REVISION);
296 write_le_32(buf + ATFS_OFFSET_INIT_BLOCK, 0);
297 write_le_32(buf + ATFS_OFFSET_INIT_SIZE, 0);
298 write_le_32(buf + ATFS_OFFSET_ROOT_BLOCK, 1);
299 write_le_32(buf + ATFS_OFFSET_ROOT_SIZE, 1);
300
301 write_le_32(buf + ATFS_FREE_OFFSET_NEXT, 2);
302 write_le_32(buf + ATFS_FREE_OFFSET_SIZE, 0);
303
304 RETURN_IF(disk_write(ATFS_SECTOR_BOOT, buf));
305
306 /* Root Dir */
307 memset(buf, 0, ATFS_BLOCK_SIZE);
308 RETURN_IF(disk_write(1, buf));
309
310 /* Free List */
311 write_le_32(buf + ATFS_FREE_OFFSET_NEXT, 0);
312 write_le_32(buf + ATFS_FREE_OFFSET_SIZE, size - 2);
313 RETURN_IF(disk_write(2, buf));
314 return STATUS_OK;
315}

317Status fs_fopen(const char *path, File *file)
318{
319 u8 buf[ATFS_BLOCK_SIZE];
320 DirEntry entry;
321 const char *s, *name;
322 char c;
323 u32 name_len;
324
325 /* Check path format beforehand for simplicity */
326 if(!path_valid(path))
327 {
328 return STATUS_PATH_FORMAT_INVALID;
329 }
330
331 /* Traverse directory structure starting in root directory */
332 RETURN_IF(disk_read(ATFS_SECTOR_BOOT, buf));
333 entry.StartBlock = load_le_32(buf + ATFS_OFFSET_ROOT_BLOCK);
334 entry.SizeBlocks = load_le_32(buf + ATFS_OFFSET_ROOT_SIZE);
335 s = path;
336 while(*s)
337 {
338 /* Get next part of path string */
339 name = s;
340 for(; (c = *s) && c != ATFS_DIR_SEPARATOR; ++s) {}
341 name_len = s - name;
342
343 /* Search current directory */
344 RETURN_IF(_find_entry(name, name_len, entry.StartBlock, &entry));
345
346 /* If in the middle of a path */
347 if(!*s)
348 {
349 /* Skip directory separator */
350 ++s;
351
352 /* Found entry must be a directory */
353 if(entry.Type)
354 {
355 return STATUS_NO_SUCH_FILE_OR_DIR;
356 }
357 }
358 }
359
360 /* Entry was found */
361 file->StartBlock = entry.StartBlock;
362 file->SizeBlocks = entry.SizeBlocks;
363 return STATUS_OK;
364}
