142static u8 _sd_command(u8 cmd, u32 arg)
143{
144 u8 i, response;
145 spi_xchg(0xFF);
146 spi_xchg(0x40 | cmd);
147 spi_xchg((arg >> 24) & 0xFF);
148 spi_xchg((arg >> 16) & 0xFF);
149 spi_xchg((arg >> 8) & 0xFF);
150 spi_xchg((arg >> 0) & 0xFF);
151 switch(cmd)
152 {
153 case CMD_GO_IDLE_STATE:
154 spi_xchg(0x95);
155 break;
156
157 case CMD_SEND_IF_COND:
158 spi_xchg(0x87);
159 break;
160
161 default:
162 spi_xchg(0xFF);
163 break;
164 }
165
166 for(i = 0; i < 10 && ((response = spi_xchg(0xFF)) == 0xFF); ++i) ;
167 return response;
168}

178static Status _sd_command_try(u8 cmd, u32 arg, u8 *out)
179{
180 u8 i, response;
181 RETURN_IF(spi_tx_try(0xFF));
182 RETURN_IF(spi_tx_try(0x40 | cmd));
183 RETURN_IF(spi_tx_try(arg >> 24));
184 RETURN_IF(spi_tx_try(arg >> 16));
185 RETURN_IF(spi_tx_try(arg >> 8));
186 RETURN_IF(spi_tx_try(arg));
187 switch(cmd)
188 {
189 case CMD_GO_IDLE_STATE:
190 RETURN_IF(spi_tx_try(0x95));
191 break;
192
193 case CMD_SEND_IF_COND:
194 RETURN_IF(spi_tx_try(0x87));
195 break;
196
197 default:
198 RETURN_IF(spi_tx_try(0xFF));
199 break;
200 }
201
202 for(i = 0; ; ++i)
203 {
204 RETURN_IF(spi_rx_try(&response));
205 if(response != 0xFF)
206 {
207 break;
208 }
209
210 if(i >= 10)
211 {
212 return STATUS_TIMEOUT;
213 }
214 }
215
216 *out = response;
217 return STATUS_OK;

560Status sd_read(u32 block, void *data)
561{
562 u16 i;
563 u8 *data8, response;
564
565 data8 = data;
566
567 SD_SELECT;
568
569 /* Start read */
570 RETURN_IF(_sd_command_try(CMD_READ_SINGLE_BLOCK,
571 _sd_block_addr(block), &response));
572 if(response)
573 {
574 SD_DESELECT;
575 return STATUS_FAIL;
576 }
577
578 /* Wait for ready */
579 for(i = 0; ; ++i)
580 {
581 RETURN_IF(spi_xchg_try(0xFF, &response));
582 if(response == 0xFE)
583 {
584 break;
585 }
586
587 if(i == 0xFFFF)
588 {
589 SD_DESELECT;
590 return STATUS_TIMEOUT;
591 }
592 }
593
594 /* Read data */
595 for(i = 0; i < BLOCK_SIZE; ++i)
596 {
597 if(spi_rx_try(&data8[i]))
598 {
599 return STATUS_TIMEOUT;
600 }
601 }
602
603 RETURN_IF(spi_tx_try(0xFF));
604 RETURN_IF(spi_tx_try(0xFF));
605 RETURN_IF(spi_tx_try(0xFF));
606 SD_DESELECT;
607 return STATUS_OK;
