135static void _draw_code(void)
136{
137 i32 i, j, brackets, len, line, col, keyword_remain;
138 bool in_comment, in_string, in_char;
139 char *text, c;
140 Color color;
141
142 len = vector_len(&_code);
143 text = vector_data(&_code);
144
145 in_comment = false;
146 in_string = false;
147 in_char = false;
148 keyword_remain = 0;
149
150 col = 0;
151 line = 0;
152 brackets = 0;
153
154 for(i = 0; i < len; ++i)
155 {
156 color = COLOR_WHITE;
157 c = text[i];
158
159 if(!keyword_remain && !isalnum(text[i - 1]))
160 {
161 i32 word_len;
162
163 /* Find end of word */
164 for (j = i; j < len; ++j)
165 {
166 if (!isalnum(text[j]))
167 {
168 break;
169 }
170 }
171
172 word_len = j - i;
173 if(keyword_detect(&text[i], word_len))
174 {
175 keyword_remain = word_len;
176 color = COLOR_BLUE;
177 }
178 }
179
180 if(c == '(' || c == '[' || c == '{')
181 {
182 color = _get_bracket_color(brackets);
183 ++brackets;
184 }
185 else if(c == ')' || c == ']' || c == '}')
186 {
187 --brackets;
188 color = _get_bracket_color(brackets);
189 }
190 else if(isdigit(c))
191 {
192 color = COLOR_LIME;
193 }
194
195 if(keyword_remain)
196 {
197 --keyword_remain;
198 color = COLOR_BLUE;
199 }
200
201 if(c == '\'' && text[i - 1] != '\\' && !in_string)
202 {
203 color = COLOR_ORANGE;
204 in_char = !in_char;
205 }
206
207 if(c == '"' && text[i - 1] != '\\' && !in_char)
208 {
209 color = COLOR_ORANGE;
210 in_string = !in_string;
211 }
212
213 if(c == '/' && text[i + 1] == '*')
214 {
215 color = COLOR_GREEN;
216 in_comment = true;
217 }
218
219 if(c == '/' && text[i - 1] == '*')
220 {
221 color = COLOR_GREEN;
222 in_comment = false;
223 }
224
225 if(in_char || in_string)
226 {
227 color = COLOR_ORANGE;
228 }
229
230 if(in_comment)
231 {
232 color = COLOR_GREEN;
233 }
234

236 {
237 col = 0;
238 ++line;
239 }

241 {

243
244 x = col * 8 + 33;
245 y = line * LINE_HEIGHT + 22;
246 ++col;
247 if (x <= GFX_WIDTH - 10 && y <= GFX_HEIGHT - 10)

249 font_string_len(x, y, &c, 1, ubuntu_regular, color, COLOR_BLACK);
250 }
251 }
252 }
