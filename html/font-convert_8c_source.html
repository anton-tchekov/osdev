94static i32 _sort_unique(i32 *elements, i32 count)
95{
96 i32 a, b;
97 qsort(elements, count, sizeof(*elements), _compare_i32);
98
99 /* Remove duplicates */
100 for(a = 1, b = 1; a < count; ++a)
101 {
102 if(elements[a] == elements[a - 1])
103 {
104 continue;
105 }
106
107 elements[b++] = elements[a];
108 }
109
110 return b;
111}
