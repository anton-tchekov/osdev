258static void _xmem_overlap(u32 addr, u16 size, AddrHelper *h)
259{
260 u32 addr_end;
261
262 h->AddrFirst = _addr_bank_offset(addr);
263 addr_end = addr + size;
264
265 h->BankFirst = _addr_to_bank(addr);
266 h->BankSecond = _addr_to_bank(addr_end);
267 if(h->BankFirst == h->BankSecond)
268 {
269 h->SizeFirst = size;
270 }
271 else
272 {
273 h->SizeSecond = _addr_bank_offset(addr_end);
274 h->SizeFirst = size - h->SizeSecond;
275 }
276}
