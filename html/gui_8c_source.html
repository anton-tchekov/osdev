190static void input_render(Input *i, bool sel)
191{
192 Color inner_color;
193 i32 y = TITLE_BAR_HEIGHT + i->Y;
194
195 const char *text = i->Flags & FLAG_PASSWORD ?
196 _password_stars : vector_data(&i->Text);
197
198 inner_color = sel ?
199 _current_theme->ElementSelBG :
200 _current_theme->ElementBG;
201
202 gfx_rect(i->X + BORDER_SIZE,
203 TITLE_BAR_HEIGHT + i->Y + BORDER_SIZE,
204 i->W - 2 * BORDER_SIZE,
205 INPUT_HEIGHT - 2 + BORDER_SIZE,
206 inner_color);
207
208 if(sel)
209 {
210 gfx_rect_border(i->X, TITLE_BAR_HEIGHT + i->Y,
211 i->W, INPUT_HEIGHT, BORDER_SIZE_SEL,
212 _current_theme->ColorBorderSel);
213 }
214 else
215 {
216 gfx_rect_border(i->X, y, i->W, INPUT_HEIGHT, BORDER_SIZE,
217 _current_theme->ColorBorder);
218 }
219
220 if(i->Selection == i->Position)
221 {
222 font_string_len(i->X + INPUT_PADDING_X, y + INPUT_PADDING_Y,
223 text, vector_len(&i->Text),
224 font_default,
225 _current_theme->ColorFG,
226 inner_color);
227 }
228 else
229 {
230 i32 sel_start, sel_len, sel_x;
231
232 sel_start = i32_min(i->Selection, i->Position);
233 sel_len = i32_max(i->Selection, i->Position) - sel_start;
234
235 /* Before selection */
236 font_string_len(i->X + INPUT_PADDING_X,
237 y + INPUT_PADDING_Y,
238 text, sel_start,
239 font_default,
240 _current_theme->ColorFG,
241 inner_color);
242
243 /* Selection */
244 sel_x = font_string_width_len(text, sel_start, font_default);
245
246 gfx_rect(i->X + INPUT_PADDING_X + sel_x, y + CURSOR_OFFSET_Y,
247 font_string_width_len(text + sel_start, sel_len, font_default),
248 CURSOR_HEIGHT,
249 _current_theme->ColorFG);
250
251 font_string_len(i->X + INPUT_PADDING_X + sel_x, y + INPUT_PADDING_Y,
252 text + sel_start, sel_len,
253 font_default,
254 _current_theme->ColorTextSelFG,
255 _current_theme->ColorTextSelBG);
256
257 /* After selection */
258 font_string_len(i->X + INPUT_PADDING_X +
259 font_string_width_len(text, sel_start + sel_len, font_default),
260 y + INPUT_PADDING_Y,
261 text + sel_start + sel_len,
262 vector_len(&i->Text) - sel_start - sel_len,
263 font_default,
264 _current_theme->ColorFG,
265 inner_color);
266 }
267
268 if(sel)
269 {
270 /* Cursor */
271 gfx_rect(i->X + INPUT_PADDING_X +
272 font_string_width_len(text, i->Position, font_default) +
273 CURSOR_OFFSET, TITLE_BAR_HEIGHT + i->Y + CURSOR_OFFSET_Y,
274 CURSOR_WIDTH,
275 CURSOR_HEIGHT,
276 _current_theme->ColorCursor);
277 }

698void window_event_key(Key key, i32 chr, KeyState state)
699{
700 void *ce;
701 if(!_current_window)
702 {
703 return;
704 }
705
706 if(_current_window->OnKey)
707 {
708 _current_window->OnKey(key, chr, state);
709 }
710
711 if(_current_window->Selected < 0)
712 {
713 return;
714 }
715
716 if(state == KEYSTATE_RELEASED)
717 {
718 return;
719 }
720
721 ce = _current_window->Elements[_current_window->Selected];
722 if(key == KEY_TAB)
723 {
724 element_next();
725 }
726 else if(key == (KEY_TAB | MOD_SHIFT))
727 {
728 element_prev();
729 }
730 else
731 {
732 ElementType type = ((Element *)ce)->Type;
733 if(type == ELEMENT_TYPE_BUTTON)
734 {
735 if(key == KEY_RETURN)
736 {
737 Button *b = ce;
738 if(b->Click)
739 {
740 b->Click();
741 }
742 }
743 }
744 else if(type == ELEMENT_TYPE_INPUT)
745 {
746 input_event_key((Input *)ce, key, chr);
747 }
748 }
749}
