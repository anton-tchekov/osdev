101static void label_render(Label *l)
102{
103 u32 align, w = 0, y, rx = 0, h = 12 + 2 * LABEL_Y_PADDING;
104 Color fg, bg;
105
106 y = TITLE_BAR_HEIGHT + l->Y;
107 align = l->Flags & ALIGN_MASK;
108
109 if (l->Flags & FLAG_INVERTED)
110 {
111 fg = _current_theme->ColorBG;
112 bg = _current_theme->ColorFG;
113 }
114 else
115 {
116 fg = _current_theme->ColorFG;
117 bg = _current_theme->ColorBG;
118 }
119
120 if (align == FLAG_ALIGN_CENTER)
121 {
122 w = font_string_width(l->Text, font_default) + 2 * LABEL_X_PADDING;
123 rx = w / 2;
124 }
125 else if (align == FLAG_ALIGN_RIGHT)
126 {
127 w = font_string_width(l->Text, font_default) + 2 * LABEL_X_PADDING;
128 rx = w;
129 }
130 else if (align == FLAG_ALIGN_LEFT)
131 {
132 if (l->Flags & FLAG_INVERTED)
133 {
134 w = font_string_width(l->Text, font_default) + 2 * LABEL_X_PADDING;
135 }
136 }
137
138 if (l->Flags & FLAG_INVERTED)
139 {
140 if (l->Flags & FLAG_INVISIBLE)
141 {
142 gfx_rect(l->X - rx - LABEL_X_PADDING, y - LABEL_Y_PADDING, w, h,
143 _current_theme->ColorBG);
144 return;
145 }
146 else
147 {
148 gfx_rect(l->X - rx - LABEL_X_PADDING, y - LABEL_Y_PADDING, w, h, bg);
149 }
150 }
151
152 font_string(l->X - rx, y, l->Text, font_default, fg, bg);
153}

212static void input_render(Input *i, bool sel)
213{
214 Color inner_color;
215 i32 y = TITLE_BAR_HEIGHT + i->Y;
216
217 const char *text = _input_get_display_text(i);
218
219 inner_color = sel ? _current_theme->ElementSelBG : _current_theme->ElementBG;
220
221 gfx_rect(i->X + BORDER_SIZE,
222 TITLE_BAR_HEIGHT + i->Y + BORDER_SIZE,
223 i->W - 2 * BORDER_SIZE,
224 INPUT_HEIGHT - 2 + BORDER_SIZE,
225 inner_color);
226
227 if (sel)
228 {
229 gfx_rect_border(i->X, TITLE_BAR_HEIGHT + i->Y,
230 i->W, INPUT_HEIGHT, BORDER_SIZE_SEL,
231 _current_theme->ColorBorderSel);
232 }
233 else
234 {
235 gfx_rect_border(i->X, y, i->W, INPUT_HEIGHT, BORDER_SIZE,
236 _current_theme->ColorBorder);
237 }
238
239 if (i->Selection == i->Position)
240 {
241 font_string_len(i->X + INPUT_PADDING_X, y + INPUT_PADDING_Y,
242 text, vector_len(&i->Text),
243 font_default,
244 _current_theme->ColorFG,
245 inner_color);
246 }
247 else
248 {
249 i32 sel_start, sel_len, sel_x;
250
251 sel_start = i32_min(i->Selection, i->Position);
252 sel_len = i32_max(i->Selection, i->Position) - sel_start;
253
254 /* Before selection */
255 font_string_len(i->X + INPUT_PADDING_X,
256 y + INPUT_PADDING_Y,
257 text, sel_start,
258 font_default,
259 _current_theme->ColorFG,
260 inner_color);
261
262 /* Selection */
263 sel_x = font_string_width_len(text, sel_start, font_default);
264
265 gfx_rect(i->X + INPUT_PADDING_X + sel_x, y + CURSOR_OFFSET_Y,
266 font_string_width_len(text + sel_start, sel_len, font_default),
267 CURSOR_HEIGHT,
268 _current_theme->ColorFG);
269
270 font_string_len(i->X + INPUT_PADDING_X + sel_x, y + INPUT_PADDING_Y,
271 text + sel_start, sel_len,
272 font_default,
273 _current_theme->ColorTextSelFG,
274 _current_theme->ColorTextSelBG);
275
276 /* After selection */
277 font_string_len(i->X + INPUT_PADDING_X +
278 font_string_width_len(text, sel_start + sel_len, font_default),
279 y + INPUT_PADDING_Y,
280 text + sel_start + sel_len,
281 vector_len(&i->Text) - sel_start - sel_len,
282 font_default,
283 _current_theme->ColorFG,
284 inner_color);
285 }
286
287 if (sel)
288 {
289 /* Cursor */
290 gfx_rect(i->X + INPUT_PADDING_X +
291 font_string_width_len(text, i->Position, font_default) +
292 CURSOR_OFFSET,
293 TITLE_BAR_HEIGHT + i->Y + CURSOR_OFFSET_Y,
294 CURSOR_WIDTH,
295 CURSOR_HEIGHT,
296 _current_theme->ColorCursor);
297 }
298}

314static void input_selection_replace(Input *i, const char *str, i32 len)
315{
316 const char *text;
317 i32 sel_start, sel_len, w_new, w_start, w_end;
318
319 sel_start = i32_min(i->Selection, i->Position);
320 sel_len = i32_max(i->Selection, i->Position) - sel_start;
321
322 text = _input_get_display_text(i);
323
324 w_new = font_string_width_len(str, len, font_default);
325 w_start = font_string_width_len(text, sel_start, font_default);
326 w_end = font_string_width_len(text + sel_start + sel_len,
327 vector_len(&i->Text) - sel_start - sel_len, font_default);
328
329 if (w_new + w_start + w_end >= i->W - 2 * INPUT_PADDING_X)
330 {
331 return;
332 }
333
334 vector_replace(&i->Text, sel_start, sel_len, str, len);
335 i->Position = sel_start + len;
336 i->Selection = i->Position;
337 input_render(i, true);
338}

803void window_event_key(Key key, i32 chr, KeyState state)
804{
805 void *ce;
806 if (!_current_window)
807 {
808 return;
809 }
810
811 if (_current_window->OnKey)
812 {
813 _current_window->OnKey(key, chr, state);
814 }
815
816 if (_current_window->Selected < 0)
817 {
818 return;
819 }
820
821 if (state == KEYSTATE_RELEASED)
822 {
823 return;
824 }
825
826 ce = _current_window->Elements[_current_window->Selected];
827 if (key == KEY_TAB)
828 {
829 element_next();
830 }
831 else if (key == (KEY_TAB | MOD_SHIFT))
832 {
833 element_prev();
834 }
835 else
836 {
837 ElementType type = ((Element *)ce)->Type;
838 if (type == ELEMENT_TYPE_BUTTON)
839 {
840 if (key == KEY_RETURN)
841 {
842 Button *b = ce;
843 if (b->Click)
844 {
845 b->Click();
846 }
847 }
848 }
849 else if (type == ELEMENT_TYPE_INPUT)
850 {
851 input_event_key((Input *)ce, key, chr);
852 }
853 }
