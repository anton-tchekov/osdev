436static u8 syscall_serial_write(u32 *args)
437{
438 u8 buf[128];
439 u16 cur;
440 u32 ptr, len;
441
442 ptr = args[0];
443 len = args[1];
444 if(_memory_check_bounds(ptr, len))
445 {
446 return 1;
447 }
448
449 while(len)
450 {
451 cur = len > sizeof(buf) ? sizeof(buf) : len;
452 env_memory_read(ptr, buf, cur);
453 env_serial_write(buf, cur);
454 len -= cur;
455 ptr += cur;
456 }
457
458 return 0;
459}

655static u8 emulator_next(Emulator *emu)
656{
657 u32 instr, opcode;
658 if(memory_lw(emu->PC, &instr))
659 {
660 return 1;
661 }
662
663 /* r0 = Zero Register */
664 emu->Registers[0] = 0;
665
666 /* bits 7 .. 0 */
667 opcode = instr & 0x7F;
668 if((opcode & 0x03) != 0x03)
669 {
670 EMU_LOG(PSTR("INVALID"));
671 return 1;
672 }
673
674 opcode >>= 2;
675 switch(opcode)
676 {
677 case OPCODE_ECALL:
678 if(syscall(emu->Registers[17], &emu->Registers[10]))
679 {
680 EMU_LOG(PSTR("SYSCALL ERROR"));
681 return 1;
682 }
683 break;
684
685 case OPCODE_LOAD:
686 {
687 i32 offset;
688 u32 imm;
689 u8 rd, rs1, funct3;
690 rd = (instr >> 7) & 0x1F;
691 funct3 = (instr >> 12) & 0x07;
692 rs1 = (instr >> 15) & 0x1F;
693 imm = (instr >> 20);
694 offset = sext(12, imm);
695 switch(funct3)
696 {
697 case 0:
698 /* LB */
699 EMU_LOG(PSTR("lb r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
700 if(memory_lb(emu->Registers[rs1] + offset, &emu->Registers[rd]))
701 {
702 return 1;
703 }
704 break;
705
706 case 1:
707 /* LH */
708 EMU_LOG(PSTR("lh r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
709 if(memory_lh(emu->Registers[rs1] + offset, &emu->Registers[rd]))
710 {
711 return 1;
712 }
713 break;
714
715 case 2:
716 /* LW */
717 EMU_LOG(PSTR("lw r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
718 if(memory_lw(emu->Registers[rs1] + offset, &emu->Registers[rd]))
719 {
720 return 1;
721 }
722 break;
723
724 case 4:
725 /* LBU */
726 EMU_LOG(PSTR("lbu r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
727 if(memory_lbu(emu->Registers[rs1] + offset, &emu->Registers[rd]))
728 {
729 return 1;
730 }
731 break;
732
733 case 5:
734 /* LHU */
735 EMU_LOG(PSTR("lhu r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
736 if(memory_lhu(emu->Registers[rs1] + offset, &emu->Registers[rd]))
737 {
738 return 1;
739 }
740 break;
741
742 default:
743 EMU_LOG(PSTR("INVALID"));
744 return 1;
745 }
746 break;
747 }
748
749 case OPCODE_IMM:
750 {
751 u8 rd, rs1, funct3;
752 u32 imm;
753 rd = (instr >> 7) & 0x1F;
754 funct3 = (instr >> 12) & 0x07;
755 rs1 = (instr >> 15) & 0x1F;
756 imm = (instr >> 20);
757 switch(funct3)
758 {
759 case 0:
760 /* ADDI */
761 imm = sext(12, imm);
762 EMU_LOG(PSTR("addi r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
763 emu->Registers[rd] = emu->Registers[rs1] + imm;
764 break;
765
766 case 1:
767 /* SLLI */
768 imm &= 0x1F;
769 EMU_LOG(PSTR("slli r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
770 emu->Registers[rd] = emu->Registers[rs1] << imm;
771 break;
772
773 case 2:
774 /* SLTI */
775 imm = sext(12, imm);
776 EMU_LOG(PSTR("slti r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
777 emu->Registers[rd] = (i32)emu->Registers[rs1] < (i32)imm;
778 break;
779
780 case 3:
781 /* SLTIU */
782 EMU_LOG(PSTR("sltiu r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
783 emu->Registers[rd] = emu->Registers[rs1] < imm;
784 break;
785
786 case 4:
787 /* XORI */
788 imm = sext(12, imm);
789 EMU_LOG(PSTR("xori r%"PRIu8", r%"PRIu8", 0x%08"PRIx32), rd, rs1, imm);
790 emu->Registers[rd] = emu->Registers[rs1] ^ imm;
791 break;
792
793 case 5:
794 if(imm >> 5)
795 {
796 /* SRAI */
797 imm &= 0x1F;
798 EMU_LOG(PSTR("srai r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
799 emu->Registers[rd] = (i32)emu->Registers[rs1] >> imm;
800 }
801 else
802 {
803 /* SRLI */
804 imm &= 0x1F;
805 EMU_LOG(PSTR("srli r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
806 emu->Registers[rd] = emu->Registers[rs1] >> imm;
807 }
808 break;
809
810 case 6:
811 /* ORI */
812 imm = sext(12, imm);
813 EMU_LOG(PSTR("ori r%"PRIu8", r%"PRIu8", 0x%08"PRIx32), rd, rs1, imm);
814 emu->Registers[rd] = emu->Registers[rs1] | imm;
815 break;
816
817 case 7:
818 /* ANDI */
819 imm = sext(12, imm);
820 EMU_LOG(PSTR("andi r%"PRIu8", r%"PRIu8", 0x%08"PRIx32), rd, rs1, imm);
821 emu->Registers[rd] = emu->Registers[rs1] & imm;
822 break;
823 }
824 break;
825 }
826
827 case OPCODE_AUIPC:
828 {
829 u8 rd;
830 u32 imm;
831 rd = (instr >> 7) & 0x1F;
832 imm = (instr >> 12);
833 EMU_LOG(PSTR("auipc r%"PRIu8", 0x%08"PRIx32), rd, imm);
834 emu->Registers[rd] = emu->PC + (imm << 12);
835 break;
836 }
837
838 case OPCODE_STORE:
839 {
840 i32 offset;
841 u32 imm;
842 u8 rs1, rs2, funct3;
843 funct3 = (instr >> 12) & 0x07;
844 rs1 = (instr >> 15) & 0x1F;
845 rs2 = (instr >> 20) & 0x1F;
846 imm = ((instr & 0xF80) >> 7) | ((instr & 0xFE000000) >> 20);
847 offset = sext(12, imm);
848 switch(funct3)
849 {
850 case 0:
851 /* SB */
852 EMU_LOG(PSTR("sb r%"PRIu8"%+"PRIi32", r%"PRIu8), rs1, offset, rs2);
853 if(memory_sb(emu->Registers[rs1] + offset, emu->Registers[rs2]))
854 {
855 return 1;
856 }
857 break;
858
859 case 1:
860 /* SH */
861 EMU_LOG(PSTR("sh r%"PRIu8"%+"PRIi32", r%"PRIu8), rs1, offset, rs2);
862 if(memory_sh(emu->Registers[rs1] + offset, emu->Registers[rs2]))
863 {
864 return 1;
865 }
866 break;
867
868 case 2:
869 /* SW */
870 EMU_LOG(PSTR("sw r%"PRIu8"%+"PRIi32", r%"PRIu8), rs1, offset, rs2);
871 if(memory_sw(emu->Registers[rs1] + offset, emu->Registers[rs2]))
872 {
873 return 1;
874 }
875 break;
876
877 default:
878 EMU_LOG(PSTR("INVALID"));
879 return 1;
880 }
881 break;
882 }
883
884 case OPCODE_REG:
885 {
886 u8 rd, rs1, rs2, funct3, funct7;
887 rd = (instr >> 7) & 0x1F;
888 funct3 = (instr >> 12) & 0x07;
889 rs1 = (instr >> 15) & 0x1F;
890 rs2 = (instr >> 20) & 0x1F;
891 funct7 = (instr >> 25);
892 if(funct7 == 1)
893 {
894 /* Multiplication Extension */
895 switch(funct3)
896 {
897 case 0:
898 /* MUL */
899 EMU_LOG(PSTR("mul r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
900 emu->Registers[rd] = (i32)emu->Registers[rs1] * (i32)emu->Registers[rs2];
901 break;
902
903 case 1:
904 /* MULH */
905 EMU_LOG(PSTR("mulh r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
906 emu->Registers[rd] = ((i64)emu->Registers[rs1] * (i64)emu->Registers[rs2]) >> 32;
907 break;
908
909 case 2:
910 /* MULHSU */
911 EMU_LOG(PSTR("mulhsu r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
912 emu->Registers[rd] = ((i64)emu->Registers[rs1] * (u64)emu->Registers[rs2]) >> 32;
913 break;
914
915 case 3:
916 /* MULHU */
917 EMU_LOG(PSTR("mulhu r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
918 emu->Registers[rd] = ((u64)emu->Registers[rs1] * (u64)emu->Registers[rs2]) >> 32;
919 break;
920
921 case 4:
922 /* DIV */
923 EMU_LOG(PSTR("div r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
924 emu->Registers[rd] = (i32)emu->Registers[rs1] / (i32)emu->Registers[rs2];
925 break;
926
927 case 5:
928 /* DIVU */
929 EMU_LOG(PSTR("divu r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
930 emu->Registers[rd] = emu->Registers[rs1] / emu->Registers[rs2];
931 break;
932
933 case 6:
934 /* REM */
935 EMU_LOG(PSTR("rem r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
936 emu->Registers[rd] = (i32)emu->Registers[rs1] % (i32)emu->Registers[rs2];
937 break;
938
939 case 7:
940 /* REMU */
941 EMU_LOG(PSTR("remu r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
942 emu->Registers[rd] = emu->Registers[rs1] % emu->Registers[rs2];
943 break;
944 }
945 }
946 else
947 {
948 switch(funct3)
949 {
950 case 0:
951 if(funct7)
952 {
953 /* SUB */
954 EMU_LOG(PSTR("sub r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
955 emu->Registers[rd] = emu->Registers[rs1] - emu->Registers[rs2];
956 }
957 else
958 {
959 /* ADD */
960 EMU_LOG(PSTR("add r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
961 emu->Registers[rd] = emu->Registers[rs1] + emu->Registers[rs2];
962 }
963 break;
964
965 case 1:
966 /* SLL */
967 EMU_LOG(PSTR("sll r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
968 emu->Registers[rd] = emu->Registers[rs1] << (emu->Registers[rs2] & 0x1F);
969 break;
970
971 case 2:
972 /* SLT */
973 EMU_LOG(PSTR("slt r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
974 emu->Registers[rd] = (i32)emu->Registers[rs1] < (i32)emu->Registers[rs2];
975 break;
976
