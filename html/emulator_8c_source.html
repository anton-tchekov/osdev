682static u8 emulator_next(Emulator *emu)
683{
684 u32 instr, opcode;
685 if(memory_lw(emu->PC, &instr))
686 {
687 return 1;
688 }
689
690 /* r0 = Zero Register */
691 emu->Registers[0] = 0;
692
693 /* bits 7 .. 0 */
694 opcode = instr & 0x7F;
695 if((opcode & 0x03) != 0x03)
696 {
697 EMU_LOG(PSTR("INVALID"));
698 return 1;
699 }
700
701 opcode >>= 2;
702 switch(opcode)
703 {
704 case OPCODE_ECALL:
705 if(syscall(emu->Registers[17], &emu->Registers[10]))
706 {
707 EMU_LOG(PSTR("SYSCALL ERROR"));
708 return 1;
709 }
710 break;
711
712 case OPCODE_LOAD:
713 {
714 i32 offset;
715 u32 imm;
716 u8 rd, rs1, funct3;
717 rd = (instr >> 7) & 0x1F;
718 funct3 = (instr >> 12) & 0x07;
719 rs1 = (instr >> 15) & 0x1F;
720 imm = (instr >> 20);
721 offset = sext(12, imm);
722 switch(funct3)
723 {
724 case 0:
725 /* LB */
726 EMU_LOG(PSTR("lb r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
727 if(memory_lb(emu->Registers[rs1] + offset, &emu->Registers[rd]))
728 {
729 return 1;
730 }
731 break;
732
733 case 1:
734 /* LH */
735 EMU_LOG(PSTR("lh r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
736 if(memory_lh(emu->Registers[rs1] + offset, &emu->Registers[rd]))
737 {
738 return 1;
739 }
740 break;
741
742 case 2:
743 /* LW */
744 EMU_LOG(PSTR("lw r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
745 if(memory_lw(emu->Registers[rs1] + offset, &emu->Registers[rd]))
746 {
747 return 1;
748 }
749 break;
750
751 case 4:
752 /* LBU */
753 EMU_LOG(PSTR("lbu r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
754 if(memory_lbu(emu->Registers[rs1] + offset, &emu->Registers[rd]))
755 {
756 return 1;
757 }
758 break;
759
760 case 5:
761 /* LHU */
762 EMU_LOG(PSTR("lhu r%"PRIu8", r%"PRIu8"%+"PRIi32), rd, rs1, offset);
763 if(memory_lhu(emu->Registers[rs1] + offset, &emu->Registers[rd]))
764 {
765 return 1;
766 }
767 break;
768
769 default:
770 EMU_LOG(PSTR("INVALID"));
771 return 1;
772 }
773 break;
774 }
775
776 case OPCODE_IMM:
777 {
778 u8 rd, rs1, funct3;
779 u32 imm;
780 rd = (instr >> 7) & 0x1F;
781 funct3 = (instr >> 12) & 0x07;
782 rs1 = (instr >> 15) & 0x1F;
783 imm = (instr >> 20);
784 switch(funct3)
785 {
786 case 0:
787 /* ADDI */
788 imm = sext(12, imm);
789 EMU_LOG(PSTR("addi r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
790 emu->Registers[rd] = emu->Registers[rs1] + imm;
791 break;
792
793 case 1:
794 /* SLLI */
795 imm &= 0x1F;
796 EMU_LOG(PSTR("slli r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
797 emu->Registers[rd] = emu->Registers[rs1] << imm;
798 break;
799
800 case 2:
801 /* SLTI */
802 imm = sext(12, imm);
803 EMU_LOG(PSTR("slti r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
804 emu->Registers[rd] = (i32)emu->Registers[rs1] < (i32)imm;
805 break;
806
807 case 3:
808 /* SLTIU */
809 EMU_LOG(PSTR("sltiu r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
810 emu->Registers[rd] = emu->Registers[rs1] < imm;
811 break;
812
813 case 4:
814 /* XORI */
815 imm = sext(12, imm);
816 EMU_LOG(PSTR("xori r%"PRIu8", r%"PRIu8", 0x%08"PRIx32), rd, rs1, imm);
817 emu->Registers[rd] = emu->Registers[rs1] ^ imm;
818 break;
819
820 case 5:
821 if(imm >> 5)
822 {
823 /* SRAI */
824 imm &= 0x1F;
825 EMU_LOG(PSTR("srai r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
826 emu->Registers[rd] = (i32)emu->Registers[rs1] >> imm;
827 }
828 else
829 {
830 /* SRLI */
831 imm &= 0x1F;
832 EMU_LOG(PSTR("srli r%"PRIu8", r%"PRIu8", %"PRIi32), rd, rs1, imm);
833 emu->Registers[rd] = emu->Registers[rs1] >> imm;
834 }
835 break;
836
837 case 6:
838 /* ORI */
839 imm = sext(12, imm);
840 EMU_LOG(PSTR("ori r%"PRIu8", r%"PRIu8", 0x%08"PRIx32), rd, rs1, imm);
841 emu->Registers[rd] = emu->Registers[rs1] | imm;
842 break;
843
844 case 7:
845 /* ANDI */
846 imm = sext(12, imm);
847 EMU_LOG(PSTR("andi r%"PRIu8", r%"PRIu8", 0x%08"PRIx32), rd, rs1, imm);
848 emu->Registers[rd] = emu->Registers[rs1] & imm;
849 break;
850 }
851 break;
852 }
853
854 case OPCODE_AUIPC:
855 {
856 u8 rd;
857 u32 imm;
858 rd = (instr >> 7) & 0x1F;
859 imm = (instr >> 12);
860 EMU_LOG(PSTR("auipc r%"PRIu8", 0x%08"PRIx32), rd, imm);
861 emu->Registers[rd] = emu->PC + (imm << 12);
862 break;
863 }
864
865 case OPCODE_STORE:
866 {
867 i32 offset;
868 u32 imm;
869 u8 rs1, rs2, funct3;
870 funct3 = (instr >> 12) & 0x07;
871 rs1 = (instr >> 15) & 0x1F;
872 rs2 = (instr >> 20) & 0x1F;
873 imm = ((instr & 0xF80) >> 7) | ((instr & 0xFE000000) >> 20);
874 offset = sext(12, imm);
875 switch(funct3)
876 {
877 case 0:
878 /* SB */
879 EMU_LOG(PSTR("sb r%"PRIu8"%+"PRIi32", r%"PRIu8), rs1, offset, rs2);
880 if(memory_sb(emu->Registers[rs1] + offset, emu->Registers[rs2]))
881 {
882 return 1;
883 }
884 break;
885
886 case 1:
887 /* SH */
888 EMU_LOG(PSTR("sh r%"PRIu8"%+"PRIi32", r%"PRIu8), rs1, offset, rs2);
889 if(memory_sh(emu->Registers[rs1] + offset, emu->Registers[rs2]))
890 {
891 return 1;
892 }
893 break;
894
895 case 2:
896 /* SW */
897 EMU_LOG(PSTR("sw r%"PRIu8"%+"PRIi32", r%"PRIu8), rs1, offset, rs2);
898 if(memory_sw(emu->Registers[rs1] + offset, emu->Registers[rs2]))
899 {
900 return 1;
901 }
902 break;
903
904 default:
905 EMU_LOG(PSTR("INVALID"));
906 return 1;
907 }
908 break;
909 }
910
911 case OPCODE_REG:
912 {
913 u8 rd, rs1, rs2, funct3, funct7;
914 rd = (instr >> 7) & 0x1F;
915 funct3 = (instr >> 12) & 0x07;
916 rs1 = (instr >> 15) & 0x1F;
917 rs2 = (instr >> 20) & 0x1F;
918 funct7 = (instr >> 25);
919 if(funct7 == 1)
920 {
921 /* Multiplication Extension */
922 switch(funct3)
923 {
924 case 0:
925 /* MUL */
926 EMU_LOG(PSTR("mul r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
927 emu->Registers[rd] = (i32)emu->Registers[rs1] * (i32)emu->Registers[rs2];
928 break;
929
930 case 1:
931 /* MULH */
932 EMU_LOG(PSTR("mulh r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
933 emu->Registers[rd] = ((i64)emu->Registers[rs1] * (i64)emu->Registers[rs2]) >> 32;
934 break;
935
936 case 2:
937 /* MULHSU */
938 EMU_LOG(PSTR("mulhsu r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
939 emu->Registers[rd] = ((i64)emu->Registers[rs1] * (u64)emu->Registers[rs2]) >> 32;
940 break;
941
942 case 3:
943 /* MULHU */
944 EMU_LOG(PSTR("mulhu r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
945 emu->Registers[rd] = ((u64)emu->Registers[rs1] * (u64)emu->Registers[rs2]) >> 32;
946 break;
947
948 case 4:
949 /* DIV */
950 EMU_LOG(PSTR("div r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
951 emu->Registers[rd] = (i32)emu->Registers[rs1] / (i32)emu->Registers[rs2];
952 break;
953
954 case 5:
955 /* DIVU */
956 EMU_LOG(PSTR("divu r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
957 emu->Registers[rd] = emu->Registers[rs1] / emu->Registers[rs2];
958 break;
959
960 case 6:
961 /* REM */
962 EMU_LOG(PSTR("rem r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
963 emu->Registers[rd] = (i32)emu->Registers[rs1] % (i32)emu->Registers[rs2];
964 break;
965
966 case 7:
967 /* REMU */
968 EMU_LOG(PSTR("remu r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
969 emu->Registers[rd] = emu->Registers[rs1] % emu->Registers[rs2];
970 break;
971 }
972 }
973 else
974 {
975 switch(funct3)
976 {
977 case 0:
978 if(funct7)
979 {
980 /* SUB */
981 EMU_LOG(PSTR("sub r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
982 emu->Registers[rd] = emu->Registers[rs1] - emu->Registers[rs2];
983 }
984 else
985 {
986 /* ADD */
987 EMU_LOG(PSTR("add r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
988 emu->Registers[rd] = emu->Registers[rs1] + emu->Registers[rs2];
989 }
990 break;
991
992 case 1:
993 /* SLL */
994 EMU_LOG(PSTR("sll r%"PRIu8", r%"PRIu8", r%"PRIu8), rd, rs1, rs2);
995 emu->Registers[rd] = emu->Registers[rs1] << (emu->Registers[rs2] & 0x1F);
996 break;
997
