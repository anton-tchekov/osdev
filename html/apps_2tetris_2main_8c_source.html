180static PieceType _new_piece(Piece *p)
181{
182 static PieceType next, bag[7];
183 static i32 idx = 6;
184
185 if(idx == 6)
186 {
187 /* create a new shuffled bag of pieces */
188 int i, j, tmp;
189 for(i = 0; i < 7; ++i)
190 {
191 bag[i] = i;
192 }
193
194 for(i = 0; i < 7; ++i)
195 {
196 j = rand() % 7;
197 tmp = bag[i];
198 bag[i] = bag[j];
199 bag[j] = tmp;
200 }
201
202 idx = 0;
203 }
204
205 p->X = 4;
206 p->Y = 0;
207 p->Rotation = 0;
208 p->Type = next;
209 next = bag[idx++];
210 return next;

213static bool _valid_position(FieldType *field, Piece *p)
214{
215 i32 bit, row, col;
216 u16 blocks;
217
218 row = 0;
219 col = 0;
220 blocks = Pieces[p->Type].Blocks[p->Rotation];
221
222 if(p->Y < 0)
223 {
224 return false;
225 }
226
227 for(bit = 0x8000; bit > 0; bit >>= 1)
228 {
229 if((blocks & bit) && _field_get(field, p->X + col, p->Y + row))
230 {
231 return true;
232 }
233
234 if(++col == 4)
235 {
236 col = 0;
237 ++row;
238 }
239 }
240
241 return false;

244static void _to_field(FieldType *field, Piece *p)
245{
246 i32 x, y, bit, row, col;
247 u16 blocks;
248
249 row = col = 0;

251 for(bit = 0x8000; bit > 0; bit >>= 1)
252 {
253 if(blocks & bit)
254 {
255 x = p->X + col;
256 y = p->Y + row;

258 field[y * FIELD_WIDTH + x] = p->Type + 1;
259 }

261 if(++col == 4)

300void event_key(Key key, i32 chr, KeyState state)
301{
302 if(state == KEYSTATE_RELEASED)
