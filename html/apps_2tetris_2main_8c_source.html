213static PieceType _new_piece(Piece *p)
214{
215 static PieceType next, bag[7];
216 static i32 idx = 6;
217
218 if(idx == 6)
219 {
220 /* create a new shuffled bag of pieces */
221 int i, j, tmp;
222 for(i = 0; i < 7; ++i)
223 {
224 bag[i] = i;
225 }
226
227 for(i = 0; i < 7; ++i)
228 {
229 j = rand() % 7;
230 tmp = bag[i];
231 bag[i] = bag[j];
232 bag[j] = tmp;
233 }

238 p->X = 4;
239 p->Y = 0;
240 p->Rotation = 0;
241 p->Type = next;
242 next = bag[idx++];

254static bool _valid_position(FieldType *field, Piece *p)
255{
256 i32 bit, row, col;
257 u16 blocks;
258
259 row = 0;
260 col = 0;
261 blocks = Pieces[p->Type].Blocks[p->Rotation];
262
263 for(bit = 0x8000; bit > 0; bit >>= 1)
264 {
265 if((blocks & bit) && _field_get(field, p->X + col, p->Y + row))
266 {
267 return false;
268 }
269
270 if(++col == 4)
271 {
272 col = 0;
273 ++row;
274 }
275 }
276
277 return true;
