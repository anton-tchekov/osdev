216static PieceType _new_piece(Piece *p)
217{
218 static PieceType next, bag[7];
219 static i32 idx = 6;
220
221 if(idx == 6)
222 {
223 /* create a new shuffled bag of pieces */
224 int i, j, tmp;
225 for(i = 0; i < 7; ++i)
226 {
227 bag[i] = i;
228 }
229
230 for(i = 0; i < 7; ++i)
231 {
232 j = rand() % 7;
233 tmp = bag[i];
234 bag[i] = bag[j];
235 bag[j] = tmp;
236 }

241 p->X = 4;
242 p->Y = 0;
243 p->Rotation = 0;
244 p->Type = next;
245 next = bag[idx++];

257static bool _valid_position(FieldType *field, Piece *p)
258{
259 i32 bit, row, col;
260 u16 blocks;
261
262 row = 0;
263 col = 0;
264 blocks = Pieces[p->Type].Blocks[p->Rotation];
265
266 for(bit = 0x8000; bit > 0; bit >>= 1)
267 {
268 if((blocks & bit) && _field_get(field, p->X + col, p->Y + row))
269 {
270 return false;
271 }
272
273 if(++col == 4)
274 {
275 col = 0;
276 ++row;
277 }
278 }
279
280 return true;

433void loop(void)
434{
435 static u32 last_ticks, score;
436 u32 ticks;
437
438 if((ticks = millis()) > last_ticks + _ticks_update)
439 {
440 last_ticks = ticks;
